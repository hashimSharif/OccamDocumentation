120  /*! The OrContract::emitEntryTest routine calls the \e emitEntryTest for each of the previrt::FunctionContract in the vector. */
121  virtual const BasicBlock*
122  emitEntryTest(Function* inside, const BasicBlock* success,
123  const BasicBlock* failure) const
124  {
125  const BasicBlock* top = failure;
126  for (std::vector<FunctionContract*>::const_reverse_iterator i =
127  this->contracts.rbegin(), e = this->contracts.rend(); i != e; ++i) {

134  /*! The SingletonTypeContract defines the arguments and methods to add entry tests to functions.
135  The class data member \e params is a vector of previrt::PrevirtType arguments and the method SingletonTypeContract::emitEntryTest
136  adds an entry test to a function,(testing equality of each previrt::PrevirtType argument against the corresponding function parameters)
137  */
138  struct SingletonTypeContract : public FunctionContract
139  {
140  private:
141  /// A set of previrt::PrevirtType arguments. For each argument an entry test is added to the function entry
142  const std::vector<PrevirtType> params;
143  public:
144  /// Constructor, takes as argument a vector of \e params (set of Previrt arguments for which entry test is required)
145  SingletonTypeContract(const std::vector<PrevirtType>& _params) :
146  params(_params)
147  {
148  }
149 
150  /// Default destructor
151  virtual
152  ~SingletonTypeContract()
153  {

156  /*!
157  The method adds an entry test to a function,(testing equality of each previrt::PrevirtType argument against the corresponding function parameters)
158  */
159  virtual const BasicBlock*
160  emitEntryTest(Function* inside, const BasicBlock* success,
161  const BasicBlock* failure) const
162  {
163  LLVMContext& ctx = success->getContext();
164  const BasicBlock* top = success; // The basic block is named to, as it points to the first basic block
165  Function::ArgumentListType::const_reverse_iterator args =
166  inside->getArgumentList().rbegin();
167 
168  // Looping for each of the arguments in the argument list of the function, as well as the PrevirtType arguments
169  for (std::vector<PrevirtType>::const_reverse_iterator b =
170  this->params.rbegin(), e = this->params.rend(); b != e; ++b, ++args)
171  {
172  // For each concrete parameter an entry Test is emitted
173  if (!b->isConcrete())
174  {
175  // Nothing to check for non-concrete values
176  continue;
177  }
178 
179  // Generate a function to test the equality, The equality function for int,string are different and hence the call
180  // to b->getEqualityFunction will return different equality functions for different Types of the PrevirtType object \e b
181  Function* eq = b->getEqualityFunction(inside->getParent()); // Adding an equality function in the Module if one doesn't exist
182  if (eq == NULL)
183  {
184  errs() << "No equality function, defaulting to false\n";
185  return failure;
186  }
187 
188  // Call the function
189  BasicBlock* bb = BasicBlock::Create(ctx); // Create a new basic block with the LLVMContext 'ctx'
190  inside->getBasicBlockList().push_front(bb); // Add the basic block to the start of the function 'inside'
191  IRBuilder<> builder(bb); // Creating an IR Builder for the Basic block in order to add LLVM IR
192 
193  std::vector<Value*> argVals;
194  argVals.reserve(2);
195  argVals.push_back(b->concretize(*inside->getParent(), args->getType())); // Pushing the constant (PrevirtType) to compare against the argument of the function
196  argVals.push_back(const_cast<Argument*> (&*args)); // (&*args) is the argument of the function
197  Value* test = builder.CreateCall(eq, ArrayRef<Value*> (argVals)); // Create A Call for the function with the ArgVals as arguments
198 
199  //Create a switch instruction with the specified value, default dest,
200  //and with a hint for the number of cases that will be added (for efficient allocation).
201  SwitchInst* sw = builder.CreateSwitch(test,
202  const_cast<BasicBlock*> (failure), 1); // Only adding 1 case as specified by the last argument
203  sw->addCase((ConstantInt*) ConstantInt::getTrue(Type::getInt1Ty(ctx)),
204  const_cast<BasicBlock*> (top)); // If the test is true, (the equality function returns true) jump to the success BB
205  top = bb; // Set top to the newly created BB
206  }
207  return top; // Return the first Basic Block with the entry test code inserted
208  }
209  };

233  /*!
234  * Generate a piece of code in F that enforces I, protecting calls to \e 'call'. It calls \e emitEntryTest in turn to create equality tests
235  * that validate the interface
236  */
237  void
238  protectCall(Function* F, Function* const call, const FunctionContract& I)
239  {
240  std::vector<Value*> ary;
241  ary.reserve(F->getArgumentList().size());
242  for (Function::ArgumentListType::iterator i = F->getArgumentList().begin(),
243  e = F->getArgumentList().end(); i != e; ++i) {
244  ary.push_back(&*i);
245  }
246 
247  BasicBlock* call_bb = BasicBlock::Create(F->getContext());
248  {
249  F->getBasicBlockList().push_back(call_bb);
250  IRBuilder<> builder(call_bb);
251 
252  if (F->getReturnType() == Type::getVoidTy(F->getContext())) {
253  CallInst* ci = builder.CreateCall(call, ArrayRef<Value*> (ary), "");
254  ci->setTailCall(true);
255  builder.CreateRetVoid();
256  } else {
257  CallInst* ci = builder.CreateCall(call, ArrayRef<Value*> (ary),
258  "delegate");
259  ci->setTailCall(true);
260  builder.CreateRet(ci);
261  }
262  }
263 
264  BasicBlock* fail_bb = BasicBlock::Create(F->getContext());
265  {
266  std::vector<Type*> arg_types;
267  arg_types.reserve(1);
268  arg_types.push_back(Type::getInt8PtrTy(F->getContext()));
269 
270  FunctionType* policyErrorType = FunctionType::get(Type::getVoidTy(
271  F->getContext()), ArrayRef<Type*> (arg_types), true);
272  Constant* policyError = F->getParent()->getOrInsertFunction(
273  "policyError", policyErrorType);
274 
275  F->getBasicBlockList().push_back(fail_bb);
276  IRBuilder<> builder(fail_bb);
277 
278  Constant* name = llvm::ConstantDataArray::getString(F->getContext(), F->getName(), true);
279  GlobalVariable *gv = new GlobalVariable(*F->getParent(), name->getType(),
280  true, GlobalVariable::LinkOnceODRLinkage, name, "");
281 
282  Value* v = builder.CreateConstGEP2_32(gv, 0, 0);
283  ary.insert(ary.begin(), v);
284  builder.CreateCall(policyError, ArrayRef<Value*> (ary), "");
285  if (F->getReturnType() == Type::getVoidTy(F->getContext())) {
286  builder.CreateRetVoid();
287  } else {
288  builder.CreateRet(UndefValue::get(F->getReturnType()));
289  }
290  }
291 
292  // call_bb is the success block and fail_bb is the failure block, the success block contains the call to the Function call
293  I.emitEntryTest(F, call_bb, fail_bb);
294  }

296  /*!
297  * Enforce an interface inside a function. Code is injected into function definitions
298  to test the equality of the previrt::PrevirtType arguments against the function arguments.
299  If the tests are successful, the protected function \e F is called otherwise, the \e failure function is called
300  */
301  Function*
302  enforceInterfaceInside(Function* F, const FunctionContract& I)
303  {
304  assert(!F->isDeclaration());
305 
306  Function* wrapF = Function::Create(F->getFunctionType(), F->getLinkage(),
307  "", F->getParent()); // A wrapper function is created and added to the Module
308  wrapF->copyAttributesFrom(F);
309  F->setLinkage(Function::InternalLinkage); // Set linkage of F to internalLinkage
310 
311  // Creating a piece of code in wrapF to protect calls to Function F, implementing the interface I
312  protectCall(wrapF, F, I);
313 
314  return wrapF;
315  }

319  * Enforce an interface outside a call (For function declarations). A wrapper is created around the declaration, that tests the equality of the previrt::PrevirtType arguments against the function arguments.

337  /*! The function takes as argument a function name, and creates a vector of FunctionContract based on the usages of the function
338  * (arguments to calls). An object of previrt::OrContract is created and returned instantiated with the vector of FunctionContract
339  *
340  */
341  FunctionContract*
342  getContractFromInterface(StringRef name, ComponentInterface& T)
343  {
344  ComponentInterface::FunctionIterator f = T.find(name);
345  if (f == T.end())
346  return NULL;
347 
348  std::vector<FunctionContract*> options;
349 
350  options.reserve(f->second.size());
351  for (ComponentInterface::CallIterator i = T.call_begin(f), e =
352  T.call_end(f); i != e; ++i) {
353  options.push_back(FunctionContract::fromCallInfo(**i));
354  }
355 
356  return FunctionContract::anyContract(options);
357  }

361  /*! The enforceInterface function iterates through all the functions that are a part of the previrt::ComponentInterface
362  enforcing interfaces to the functions by calling previrt::enforceInterfaceInside for a function definition,
363  and previrt::enforceInterfaceOutside for a function declaration
364  */
365  bool
366  enforceInterface(Module& M, ComponentInterface& T, bool inside)
367  {
368  // Iterating over all of the Functions that are a part of the Component Interface.
369  // The functions on which an interface is to be imposed
370  for (ComponentInterface::FunctionIterator i = T.begin(), e = T.end(); i != e; ++i)
371  {
372  Function* f = M.getFunction(i->first()); // Given the function name, get the function pointer.
373  if (f == NULL) // If the function was not defined in the module or declared in the module
374  continue;
375 
376  if (inside) // If enforceInterfaceInside
377  {
378  if (!f->isDeclaration()) // f should not be a declaration, we expect a function definition
379  {
380  FunctionContract* contract =
381  getContractFromInterface(f->getName(), T);
382  if (contract) // If there does exist a contract for the function, in the previrt::ComponentInterface
383  {
384  enforceInterfaceInside(f, *contract);
385  delete contract;
386  }
387  }
388  }
389  else // If enforceInterfaceOutside
390  {
391  if (f->isDeclaration()) // we expect a function declaration
392  {
393  FunctionContract* contract = getContractFromInterface(f->getName(), T);
394 
395  if (contract)
396  {
397  enforceInterfaceOutside(f, *contract);
398  delete contract;
399  }
400  }
401  }
402 
403 
404  }
405  return true;
406  }
