349  int
350  PrevirtType::refines(const llvm::Value* const val) const
351  {
352  assert(val != NULL);
353  const Constant* cnst = dyn_cast<const Constant>(val);
354 
355  if (cnst == NULL)
356  {
357  if (this->buffer.type() == proto::U)
358  return LOOSE_MATCH;
359  else
360  return NO_MATCH;
361  }
362  switch (buffer.type())
363  {
364  default:
365  assert(false);
366  break;
367  case proto::U:
368  return LOOSE_MATCH;
369  case proto::N:
370  {
371  if (cnst->isNullValue())
372  return EXACT_MATCH;
373  else
374  return NO_MATCH;
375  }
376  case proto::S:
377  {
378  StringRef out;
379  if (StringFromValue(val, out))
380  {
381  if (out == buffer.str().data())
382  return EXACT_MATCH;
383  else
384  return NO_MATCH;
385  }
386  return NO_MATCH;
387  }
388  case proto::I:
389  if (const ConstantInt* va = dyn_cast<const ConstantInt>(val))
390  {
391  if (buffer.int_().bits() == va->getBitWidth()
392  && buffer.int_().value() == va->getValue().toString(16, true))
393  return EXACT_MATCH;
394  }
395  return NO_MATCH;
396  case proto::F:
397  if (const ConstantFP* va = dyn_cast<const ConstantFP>(val))
398  {
399  const fltSemantics* sem = NULL;
400  switch (buffer.float_().sem()) {
401 #define CASE(x) case proto::x: { if (&va->getValueAPF().getSemantics() != &APFloat::x) return NO_MATCH; sem = &APFloat::x; break; }
402  CASE(IEEEdouble)
403  CASE(IEEEhalf)
404  CASE(IEEEsingle)
405  CASE(IEEEquad)
406  CASE(x87DoubleExtended)
407  CASE(PPCDoubleDouble)
408  CASE(Bogus)
409 #undef CASE
410  }
411  APFloat apf(*sem, buffer.float_().data());
412  if (apf.bitwiseIsEqual(va->getValueAPF())) {
413  return EXACT_MATCH;
414  }
415  }
416  return NO_MATCH;
417  case proto::G:
418  if (const GlobalValue* gv = dyn_cast<const GlobalValue>(val->stripPointerCasts()))
419  {
420  if (gv->getName() == buffer.global().name())
421  {
422  return EXACT_MATCH;
423  }
424  }
425  return NO_MATCH;
426  }
427 
428  return NO_MATCH;
429  }

432  /*! Return a global variable containing the value of the PrevirtType */
433  llvm::Value*
434  PrevirtType::concretize(Module& M, Type* type) const
435  {
436  llvm::Value *concreteValue = NULL;
437  switch (buffer.type()) {
438  default:
439  break;
440  case proto::N:
441  concreteValue = Constant::getNullValue(type);
442  break;
443  case proto::I:
444  assert(buffer.int_().IsInitialized());
445  concreteValue =
446  ConstantInt::get(M.getContext(),
447  APInt(buffer.int_().bits(), buffer.int_().value(), 16));
448 
449  break;
450  case proto::S:
451  assert(buffer.str().IsInitialized());
452  if (!buffer.str().cstr())
453  break;
454  { // Scope sc locally
455  GlobalVariable* sc =
456  materializeStringLiteral(M, buffer.str().data().c_str());
457  concreteValue = charStarFromStringConstant(M, sc);
458  }
459  break;
460  case proto::G:
461  assert(buffer.global().IsInitialized());
462  concreteValue = M.getGlobalVariable(buffer.global().name(), false);
463  if (concreteValue == NULL)
464  {
465  // GlobalValues are always pointers and the resulting type
466  // will be a pointer to the type in the constructor, so we
467  // need to remove one of the pointer types to get the right
468  // type out. This works assuming the desired type was already
469  // a pointer, which should be the case because we only concretize
470  // arguments and you can't pass a structure or function except
471  // as a pointer.
472  assert(type->isPointerTy()
473  && "Unexpected concretization of G to non-pointer type");
474  Type * elemType = type->getContainedType(0);
475  if (elemType->isFunctionTy())
476  {
477  concreteValue = M.getFunction(buffer.global().name());
478  if (concreteValue == NULL)
479  {
480  concreteValue = Function::Create(cast<FunctionType>(elemType),
481  GlobalVariable::ExternalLinkage, buffer.global().name(), &M);
482  }
483  }
484  else
485  {
486  concreteValue = new GlobalVariable(M, elemType, buffer.global().is_const(),
487  GlobalVariable::ExternalLinkage, NULL, buffer.global().name());
488  }
489  }
490  break;
491  }
492  assert(type == concreteValue->getType() && "Concretization did not procude value of expected type!");
493  return concreteValue;
494  }

636  /*! Build an equality function to compare two values.
637  The Type of the buffer, determines whether a string or int equality function is required.
638  The \e cacheEq type keeps a map of Function * against Type *. If an equality function already exists
639  in the \e cacheEq map, it is simply returned, otherwise a new function to test the equality of two values taken as input, is
640  created and added to the cacheEq map for future reference.
641  */
642  Function*
643  PrevirtType::getEqualityFunction(Module* M) const
644  {
645  switch (buffer.type())
646  {
647  default:
648  return NULL;
649 
650  case proto::N:
651  {
652  return NULL;
653  }
654  case proto::I:
655  {
656  assert(buffer.int_().IsInitialized());
657  IntegerType* typ = Type::getIntNTy(M->getContext(), buffer.int_().bits()); //Create an integer type instance
658  EqCache::iterator i = PrevirtType::cacheEq.find(typ); // If already an equality function given the type exists
659  if (i != PrevirtType::cacheEq.end()) // If an equality function exist,no need to create one
660  {
661  return i->second; // The second field refers to the second field in the map, which is the equality function against the specific type (e.g int, string)
662  }
663  Function* f = buildIntEquality(typ, M->getContext()); // make a new 'int' equality function
664 
665  PrevirtType::cacheEq.insert(std::make_pair(typ, f)); // Add the type*, function* pair to the EqCache Map, for future lookup
666  M->getFunctionList().push_back(f); // Add the new function to the Module
667  return f;
668  }
669  // Repeat the same process above for a String
670  case proto::S:
671  {
