111  /*! The strongconnect function iterates over the call graph, and identifies all recursive functions. The recursive functions
112  are saved onto the \e sccs map. The RecursiveSpecPolicy::allowSpecialization searches the \e sccs map to check if a function is recursive.
113  If the function is found recursive, specialization on function arguments does not proceed.
114  */
115  static void
116  strongconnect(CallGraphNode* from, CallGraph& cg,
117  std::list<Function*>& stack, std::set<Function*>& stack_contents,
118  std::map<const Function*, std::pair<int, int> >& indicies, int& next_idx,
119  RecursiveGuardSpecPolicy::SCC& result)
120  {
121  //Q!. Why is the callGraph cg passed when it is not used. The stack_contents is always left empty?
122 
123  int base_idx = next_idx; // This variables saves the number of the function accessed.For the first function it is 0, for second 1 etc
124  Function* f = from->getFunction(); // get the function from the call graph node of the function
125  indicies[f] = std::pair<int, int>(next_idx, next_idx); // save a pair against the Function in the map
126  // The prototype of the pair initialization constructor is : pair ( const first_type& a, const second_type& b);
127  next_idx++;
128  stack.push_back(f); // Add the function to the end of the list
129 
130  bool self = false;
131 
132  for (CallGraphNode::iterator i = from->begin(), e = from->end(); i != e; ++i)
133  {
134  Function* fi = i->second->getFunction(); // i->second is a callgraphNode and getFunction is called on it to get the function*
135  if (fi == NULL) continue; // Might be a dynamic call
136  if (fi == f) self = true; // If a recursive call is made set self to true
137  if (indicies.find(fi) == indicies.end()) // If the now discovered function, was not discovered before
138  {
139  strongconnect(i->second, cg, stack, stack_contents, indicies, next_idx, result); // Repeat the procedure recursively
140  std::pair<int,int>& wi = indicies[fi]; // Creating an alias for the pair of the second Function(the function called), wi
141  std::pair<int,int>& vi = indicies[f]; // Creating an alias for the pair of the calling function, vi
142  vi.second = std::min(vi.second, wi.second);
143 
144  //Q! vi.second will always be the min of (vi.second, wi.second) Review required
145 
146  }
147  //Q! The else if below seems to never ever execute as the stack contents is empty?
148  else if (stack_contents.find(fi) != stack_contents.end()) // If the function was found in stack_contents
149  {
150  //Q! This region of code seems to be never accessed because stack_contents is empty and will always return stack_contents.end()
151  std::pair<int,int>& wi = indicies[fi];
152  std::pair<int,int>& vi = indicies[f];
153  vi.second = std::min(vi.second, wi.first);
154  }
155  }
156 
157  std::pair<int,int>& vi = indicies[f];
158  if (vi.first == vi.second)
159  {
160  if (stack.back() == f && !self) // The stack.back() functions returns the reference to the last element in the list
161  {
162  // If the function did not make a recursive call and did not call any function that was not already visited
163  stack.pop_back(); // The pop_back function removes the last element in the list
164  return; // Do not proceed further ! return from function
165  }
166 
167  Function* w = NULL;
168  // The following code is executed if a recursive call was made Or there were more functions on the stack, therefore stack.back()!=f
169  do
170  {
171  w = stack.back();
172  stack.pop_back();
173  result[w] = base_idx; // The recursive Functions are captured within the result array
174  } while (w != f); // Unless the calling function f is found in the stack, keep popping other functions from the list
175  }
176  // In the else case of this statement point to be noted is that the function that was pushed onto the stack was not
177  // popped off
178 
179  }

181  /*! The function iterates over the indices in the call graph and calls previrt::strongconnect to identify recursive functions
182  given a callgraph node and adds functions found recursive to the \e sccs map
183  */
184  static void
185  scc(CallGraph &cg, RecursiveGuardSpecPolicy::SCC& scc)
186  {
187  std::map<const Function*, std::pair<int, int> > indicies;
188  std::list<Function*> stack;
189  std::set<Function*> stack_contents;
190  int index = 0;
191 
192  // A callGraph contains a map Of Function* against CallGraphNode*
193  // The iterator over the call graph returns the index of a map whereas the map
194  // is a <Function*,callgraphNode*> map. i->first points to the Function and i->second points to the CallGraphNode * for that function
195  for (CallGraph::iterator i = cg.begin(), e = cg.end(); i != e; ++i)
196  {
197  if (indicies.find(i->first) == indicies.end()) // If the function is not found in the map
198  {
199  strongconnect(i->second, cg, stack, stack_contents, indicies, index, scc);
200  }
201  }
202  }

229  /*! If the function F is not found in the map of recursive functions \e sccs it returns true, otherwise returns false. */
230  bool
231  RecursiveGuardSpecPolicy::allowSpecialization(Function* F) const
232  {
233  SCC::const_iterator i = this->sccs.find(F); // The sccs array contains a list of all functions, that are recursive
234  if (i == this->sccs.end())
235  {
236  return true; // If the function is not found in the list and is therefore not a recursive function
237  }
238  else
239  {
240  errs() << "Skipping specialization of recursive function '" << F->getName() << "'\n";
241  return false;
242  }
243  }
