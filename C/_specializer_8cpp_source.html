61  /*! The function assigns a new unique name to the specialized version of the function.
62  The name is assigned respective to the parameters to the function. If the function name already
63  consisted of a full qualified name consisting of the arguments and their types as a part of their name
64  only the base name is returned, otherwise the base name is appended with a left bracket '(' and a question mark '?'
65  whereas each question mark corresponds to an argument.
66 
67  For example if the function name is : "printArr(arg1,arg2)", the \e specializeName routine returns "printArr" (base name)
68  and the arguments "arg1" and "arg2" in the vector \e args (sent as argument to the function).
69 
70  If the function name is "printArr" the function simply returns "printArr" (base name) and '?', for each argument in the vector
71  of \e args.
72  */
73  std::string
74  specializeName(Function* f, std::vector<std::string>& args)
75  {
76  args.clear();
77 
78  std::string fn = f->getName().str(); // Read in the function name
79  int idx = fn.find('(');
80  if (idx != -1 && fn[fn.length()-1] == ')') { // If the name contains paranthesis, the Following block of code executes
81  std::string base = fn.substr(0, idx); // String base reads in the function name up till the first left bracket
82  idx++;
83  unsigned int st = idx;
84  bool in_str = false;
85  // Iterating over the arguments in the function
86  while (st < fn.length()-1)
87  {
88  idx = st;
89 
90  // Each iteration of the following loop handles one argument
91  while (true)
92  {
93  if (fn[idx] == '"')
94  {
95  in_str = !in_str;
96  }
97  else if (!in_str)
98  { // Comma, Separating parameters , Not the one occurring in a string
99  if (fn[idx] == ',')
100  {
101  args.push_back(fn.substr(st, idx-st));
102  break;
103  }
104  else if (fn[idx] == ')')
105  {
106  args.push_back(fn.substr(st, idx-st)); // Push the last argument
107  break;
108  }
109  }
110  else
111  {
112  if (fn[idx] == '\\')
113  {
114  idx += 2;
115  }
116  }
117  idx++;
118  }
119  st = idx + 1;
120  }
121  return base; // return functions base name
122  }
123  else
124  { // In case the functioName does not consist of paranthesis
125  args.reserve(f->getArgumentList().size());
126  for (unsigned int i = 0; i < f->getArgumentList().size(); ++i)
127  args.push_back("?"); // Push ? for each argument
128  //iam return f->getNameStr();
129  return f->getName().str(); // return the original name of the function
130  }
131 
132  }

256  /*!
257  * Specialize a call site \e I reducing the arguments to the specialized version of the function \e nfunc, and return the new call instruction
258 
259  @param I Instruction Call site to be specialized
260  @param nfunc The pointer to the specialized function
261  @param perm Array of indices to the array of arguments in the CallInst, that have to be included in the new call-site
262  */
263  Instruction*
264  specializeCallSite(Instruction* I,
265  llvm::Function* nfunc, const std::vector<unsigned>& perm)
266  {
267 
268  // The vector "perm" is a vector of unsigned values. The values are indices into the array of arguments that are to be removed from the new call site
269  // nFunc is the specialized version of the function
270 
271  assert((nfunc->isVarArg() && nfunc->getArgumentList().size() <= perm.size())
272  || (!nfunc->isVarArg() && nfunc->getArgumentList().size() == perm.size()));
273 
274  const size_t specialized_arg_count = perm.size();
275 
276  Instruction* newInst = NULL;
277  if (CallInst* ci = dyn_cast<CallInst>(I)) // The instruction is a parameter of the function and is casted into a call instruction
278  {
279  // The instruction is cast into call instruction because only call instructions can make the getArgOperand call
280  Value** newOperands = new Value*[specialized_arg_count];
281  for (size_t j = 0; j < specialized_arg_count; ++j)
282  {
283  newOperands[j] = ci->getArgOperand(perm[j]); // Get argument operand indexed by perm passed as an argument to the function
284  // populating newOperands with only a subset of the original operands. Others are left out as a new specialized function is created
285  }
286 
287  if (ci->hasName())
288  {
289  // A new CallInst is created to call the specialized function with reduced number of arguments
290  newInst = CallInst::Create(nfunc, ArrayRef<Value*>(newOperands, specialized_arg_count), ci->getName());
291  }
292  else
293  {
294  newInst = CallInst::Create(nfunc, ArrayRef<Value*>(newOperands, specialized_arg_count));
295  }
296  delete newOperands;
297 
298  }
299 
300  else if (InvokeInst* ci = dyn_cast<InvokeInst>(I))
301  {
302 
303  Value** newOperands = new Value*[specialized_arg_count];
304  for (size_t j = 0; j < specialized_arg_count; ++j)
305  {
306  newOperands[j] = ci->getArgOperand(perm[j]);
307  }
308 
309  // Create a new invoke instruction with the reduced number of arguments
310  newInst = InvokeInst::Create(nfunc, ci->getNormalDest(),
311  ci->getUnwindDest(), ArrayRef<Value*>(newOperands,
312  specialized_arg_count), ci->getName());
313 
314  delete newOperands; // newOperands no more required, freeing memory
315 
316  }
317 
318  else
319  {
320  assert(false && "specializeCallSite got non-callsite");
321  return NULL; // Unreachable
322  }
323 
324  return newInst; // Returned the new Call instruction tailored to the specialized version of the function
325  }

328  /*! The canSpecialize function returns false if a call site within in-line assembly is found.
329  A function containing an in-line assembly call is not subject to our analysis and thereby specializing such a function
330  could remove arguments required within the in-line assembly.
331  If no call-site within in-line assembly is found, canSpecialize returns true for the function \e f
332  */
333  bool canSpecialize(Function* f)
334  {
335 
336  // The function is organized such that it makes a number of checks. If one fails the function is not specialize-able otherwise it is
337  if (f->isDeclaration())
338  return false;
339  for (inst_iterator I = inst_begin(f), E = inst_end(f); I != E; ++I) {
340  if (CallInst* ci = dyn_cast<CallInst>(&*I)) { // If the instruction is a call site
341 
342  // The call instructions made in in-line assembly are not being specialized at the moment
343  if (ci->isInlineAsm())
344  {
345  return false;
346  }
347  }
348  }
349  return true;
350  }
