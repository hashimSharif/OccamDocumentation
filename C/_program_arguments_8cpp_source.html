64  @param filename Given the filename the constructor parses the program arguments from the file and

70  SpecializeArguments::SpecializeArguments(const char* filename, const char* name) :
71  ModulePass(SpecializeArguments::ID)
72  {
73  char str[1024];
74  FILE* f = fopen(filename, "r");
75  if (f == NULL)
76  {
77  errs() << "Failed to open file '" << filename << "'\n";
78  this->argc = -1;
79  this->argv = NULL;
80  return;
81  }
82 
83  std::vector<char*> args;
84  while (fgets(str, 1024, f) != NULL)
85  {
86  int len = strlen(str);
87  while (len > 0 && (str[len - 1] == '\n' || str[len - 1] == '\r'))
88  str[--len] = '\0';
89  char* buf = new char[len + 1];
90  strcpy(buf, str);
91  args.push_back(buf);
92  }
93 
94  this->argc = args.size();
95  this->argv = new char*[this->argc];
96  int i = 0;
97 
98  errs() << "Read " << this->argc << " command line arguments:\n";
99 
100  for (std::vector<char*>::iterator itr = args.begin(), end = args.end(); itr
101  != end; ++itr, ++i)
102  {
103  this->argv[i] = *itr;
104  errs() << this->argv[i] << "\n";
105  }
106 
107  this->progName = name;
108 
109  fclose(f);
110  }

135  /*! The runOnModule method specializes the main function of the Module \e M, given the command line arguments in the input file to the pass.
136  The steps of the procedure are :
137 
138  1) Get a pointer to the main function of the Module.
139 
140  2) Create an empty new main with the same function type and linkage as the original main.
141 
142  3) The name of the original main is taken and assigned to the new main function.
143 
144  4) Create a Basic Block and add it to the new main function.
145 
146  5) For all arguments in the \e argv array (class data member) create a global variable.
147 
148  6) In the new main function, in the very first Basic Block, add an call to the original main function along with the global variable
149  arguments.
150 
151  7) Set the linkage of the original main function to GlobalValue::PrivateLinkage and a function attribute of Attribute::AlwaysInline.
152 
153  8) Apply an in-liner pass, to make embed all in-line function calls, into the function body.
154 
155  9) Thereby essentially the new main function will preserve the semantics of the body of the original function, just replacing the
156  references to \e argv and \e argc with references to global variables with static values.
157 
158  */
159  bool
160  SpecializeArguments::runOnModule(Module& M)
161  {
162  Function* f = M.getFunction("main");
163 
164  if (f == NULL)
165  {
166  errs() << "Running on module without 'main' function.\n"
167  << "Ignoring...\n";
168  return false;
169  }
170 
171  if (f->getArgumentList().size() != 2) // Error: The main function is not accepting the correct number of arguments
172  {
173  errs() << "Main module has incorrect signature\n" << f->getFunctionType();
174  return false;
175  }
176 
177  // Build the array constant
178  Type* const i32type = IntegerType::get(M.getContext(), 32); // 32 is the number of bits of storage allocated for the i32Type variable
179  // This constructs a pointer to an object of the specified type in the generic address space (address space zero)
180  Type* const stringtype = PointerType::getUnqual(IntegerType::get(M.getContext(), 8));
181 
182  Function *new_main = Function::Create(f->getFunctionType(), f->getLinkage(), "", &M);
183 
184  new_main->takeName(f); // The name of the "new_main" is set to "main" and the name of function f (original main) is set to empty string ""
185  Function::arg_iterator ai = new_main->arg_begin();
186  ai++; // Skipping the first argument of main that is argc
187  Value* argvArg = (Value*) &(*ai); // argvArg points to the char * argv[] array, that is the second argument to the main function
188 
189  BasicBlock* bb = BasicBlock::Create(M.getContext());
190  new_main->getBasicBlockList().push_front(bb); // Add the first basic block of the "new_main" function
191  IRBuilder<> irb(bb); // IR builder to construct the IR for the entry Basic Block
192  std::vector<Value*> cargs;
193  std::vector<Constant*> init;
194 
195  if (this->progName) {
196  GlobalVariable* gv = materializeStringLiteral(M, this->progName); //materializeStringLiteral returns a global variable holding the string
197  cargs.push_back(irb.CreateConstGEP2_32(gv, 0, 0)); // Add the global variable to the vector cargs
198  }
199  else
200  {
201  Value* progName = irb.CreateLoad(argvArg, false);
202  cargs.push_back(progName);
203  }
204 
205  // Iterating over each of the arguments, and creating a constant global variable for each separately
206  for (int i = 0; i < this->argc; ++i)
207  {
208  GlobalVariable* gv = materializeStringLiteral(M, this->argv[i]);
209  cargs.push_back(irb.CreateConstGEP2_32(gv, 0, 0));
210  }
211 
212  // The argv is supposed to be the new set of argument to the main function.
213  Value* argv = irb.CreateAlloca(stringtype, ConstantInt::get(i32type, // Creating a single string to hold the elements of the array
214  cargs.size(), false));
215 
216  int idx = 0;
217  //Iterating over the arguments in the vector and saving the pointer to each in the indices of argv; argv[0], argv[1] ,so on
218  for (std::vector<Value*>::iterator i = cargs.begin(), e = cargs.end(); i
219  != e; ++i, ++idx)
220  {
221  Value* argptr = irb.CreateConstGEP1_32(argv, idx);
222  irb.CreateStore(*i, argptr);
223  }
224 
225  // Create a call to the old main with the arguments, argc (cargs.size) and char * argv[] (argv)
226  Value* res = irb.CreateCall2(f, ConstantInt::getSigned(IntegerType::get(
227  M.getContext(), 32), cargs.size()), argv);
228  irb.CreateRet(res); // Return a return instruction "return internal_main"
229 
230  f->setLinkage(GlobalValue::PrivateLinkage); // The original main function is assigned private linkage
231  f->addFnAttr(Attribute::AlwaysInline); // Make the original main in-line able so that an in-line pass can be applied
232 
233  PassManager mgr;
234  mgr.add(createAlwaysInlinerPass());
235  mgr.run(M); // Adding the old main function to the body of the new main function
236 
237  return true;
238  }
