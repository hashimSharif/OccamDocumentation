123  /*! The specializeOn method, returns a Value **. Value ** is essentially a array defining the specialization policy against
124  each argument. If an argument is a constant and is can be partially evaluated into the function body, the argument is returned in the array.
125  For arguments cannot be partially evaluated into the function body, their corresponding index in the Value* array is NULL.
126  */
127  Value**
128  ConservativeSpecPolicy::specializeOn(Function* F, User::op_iterator begin, User::op_iterator end) const
129  {
130  //Note: The op_iterator is an iterator over the arguments in the call site and not the parameters in the function
131  Value** slice = NULL;
132  const unsigned int arg_count = F->getArgumentList().size();
133 
134  // The following loops over, the total number of arguments in the call site
135  for (unsigned int i = 0; i < arg_count && begin != end; ++begin, ++i)
136  {
137  Constant* cst = dyn_cast<Constant> (begin->get()); // Casting the value into a constant
138  if (SpecializationPolicy::isConstantSpecializable(cst)) // isConstantSpecializable is defined in the file SpecializationPolicy.h
139  {
140  if (slice == NULL) // If no specialize argument was found until now
141  {
142  slice = new Value*[arg_count];
143  for (unsigned int j = 0; j < i; ++j)
144  slice[j] = NULL; // Assign null values to all previous indices, as this is the first argument found specialize-able
145  }
146  slice[i] = begin->get(); // save the argument value, in the slice array
147  }
148  else if (slice != NULL)
149  {
150  slice[i] = NULL; // If the constant cannot be specialized
151  }
152  }
153 
154  return slice; // If no argument can be specialized , NULL is returned
155  }
