64  /*! Given a StringRef name of the function, or the name of the alias to the function, return the pointer to the function */
65  static Function*
66  resolveFunction(Module& M, StringRef name)
67  {
68  Function* f = M.getFunction(name);
69  if (f != NULL)
70  return f;
71  GlobalAlias* ga = M.getNamedAlias(name);
72  if (ga != NULL)
73  {
74  const GlobalValue* v = ga->resolveAliasedGlobal(true);
75  f = dyn_cast<Function>(const_cast<GlobalValue*>(v));
76  if (f != NULL)
77  {
78  errs() << "Resolved alias " << name << " to " << f->getName() << "\n";
79  return f;
80  }
81  }
82  return NULL;

87  /*!
88  * Reduce this module with respect to the given interface.
89  * - The interface suggests some of the uses of the functions,
90  * so here we can generate special versions of those functions.
91  * Generate a ComponentInterfaceTransform for clients to rewrite their
92  * code to use the new API.
93  * The previrt::specializeComponent calls previrt::specializeFunction, to create a specialized
94  version of the function corresponding to list of arguments in the transform object.
95  The new function name, along with arguments is written to the previrt::ComponentInterfaceTransform object.
96  the file will be used by the \e 'Prewrite' Pass to replace the existing call-sites with specialized call-sites.
97  The previrt::proto::ComponentInterface object is extracted from the previrt::ComponentInterfaceTransform class data member
98  */
99  bool
100  SpecializeComponent(Module& M, ComponentInterfaceTransform& T,
101  SpecializationPolicy &policy, std::list<Function*>& to_add)
102  {
103  int rewrite_count = 0;
104  const ComponentInterface& I = T.getInterface();
105  // TODO: What needs to be done?
106  // - Should try to handle strings & arrays
107  for (ComponentInterface::FunctionIterator ff = I.begin(), fe = I.end(); ff!= fe; ++ff)
108  {
109  StringRef name = ff->first();
110  Function* func = resolveFunction(M, name); // If the function doesn't belong to the module return null
111  if (func == NULL || func->isDeclaration())
112  {
113  // We don't specialize declarations because we don't own them
114  continue;
115  }
116 
117  // Iterating over all the calls made to the function 'name' from another module
118  for (ComponentInterface::CallIterator cc = I.call_begin(name), ce = I.call_end(name); cc != ce; ++cc)
119  {
120  const CallInfo* const call = *cc;
121  const unsigned arg_count = call->args.size();
122 
123  if (func->isVarArg())
124  {
125  // TODO: I don't know how to specialize variable argument functions yet
126  continue;
127  }
128  if (arg_count != func->getArgumentList().size())
129  {
130  // Not referring to this function?
131  // NOTE: I can't assert this equality because of the way that approximations occur
132  continue;
133  }
134 
135  SmallBitVector slice(arg_count);
136  bool shouldSpecialize = policy.specializeOn(func, call->args.begin(),
137  call->args.end(), slice);
138 
139  if (!shouldSpecialize) // If specialization is not required
140  continue;
141 
142  std::vector<Value*> args;
143  std::vector<unsigned> argPerm;
144  args.reserve(arg_count);
145  argPerm.reserve(slice.count());
146  for (unsigned i = 0; i < arg_count; i++)
147  {
148  if (slice.test(i))
149  {
150  Type * paramType = func->getFunctionType()->getParamType(i);
151  Value *concreteArg = call->args[i].concretize(M, paramType);
152  args.push_back(concreteArg);
153  assert(concreteArg->getType() == paramType
154  && "Specializing function with concrete argument of wrong type!");
155  }
156  else
157  {
158  args.push_back(NULL);
159  argPerm.push_back(i);
160  }
161 
162  }
163 
164  Function* nfunc = specializeFunction(func, args); // The function is specialized on the basis of the arguments to the function
165  // from another module
166  nfunc->setLinkage(GlobalValue::ExternalLinkage); // Set the linkage of the function to external linkage, so that it is not removed by DCE
167  FunctionHandle rewriteTo = nfunc->getName(); // The name assigned to the specialized version of the function
168 
169  T.rewrite(name, call, rewriteTo, argPerm); // saving the rewrites, name refers to the original function and rewriteTo is the
170  // specialized version. argPerm contains indices for call site specialization
171  // The original reference to the call is also saved
172  to_add.push_back(nfunc);
173  rewrite_count++;
174  }
175 
176  }
177  if (rewrite_count > 0)
178  {
179  errs() << rewrite_count << " pending rewrites\n";
180  }
181  return rewrite_count > 0;
182  }
183 }

200  /*! SpecializeComponentPass implements one phase of a three phase Inter-Module specialization process.
201  The steps of the process are:
202 
203  1) Gathering external references across all modules, so that the specific usages of functions outside the modules where they
204  are defined can be captured. This collection is performed by the \e 'Pinterface2' pass implemented, in the previrt::GatherInterface2Pass
205  class.
206 
207  2) Given the specific usages, serialized into a file as a previrt::proto::ComponentInterfaceTransform object, the SpecializeComponentPass
208  creates specialized versions of function in the modules where they have been defined.
209 
210  3) Once the specialized functions are created, the corresponding calls to the specialized versions, need to be rewritten,
211  taking a fewer number of arguments. The call rewriting process is accomplished by the \e 'Prewrite' Pass implemented by the
212  previrt::RewriteComponentPass class.
213 
214  SpecializeComponentPass is registered as a Pass ,with the PassManager internal database with the name \e 'Pspecialize'.
215  The Pass takes two arguments. The first argument (required argument) is the name of the file along with the \e 'Pspecialize-input' flag
216  containing a serialized previrt::proto::ComponentInterfaceTransform serialized buffer, that contains the list of functions to specialize along with the set
217  of argument values to specialize with respect to. The second argument (optional argument) is the name of an output file that follows the \e 'Pspecialize-output' flag. The uniquely assigned
218  function names of the specialized functions along with the reduced set of arguments is printed to the file as a serialized
219  previrt::proto::ComponentInterfaceTransform Google protocol buffer. The file is input to the next phase of inter-module specialization that is
220  previrt::RewriteComponentPass that takes as input the previrt::proto::ComponentInterfaceTransform buffer and specializes all call-sites
221  made to the specialized functions.
222  */

274  /*! The SpecializeComponentPass::runOnModule method creates specialized versions of the functions, corresponding to each distinct context of use.
275 
276  1) previrt::SpecializeComponent is called to create specialized functions with respect to the previrt::proto::ComponentInterfaceTransform
277  (defining the usage) of the functions.
278 
279  2) The new specialized functions created are added to the Module M.
280 
281  3) If an output file name is specified as a command line argument to the \e 'Pspecialize' Pass, the Interface consisting of
282  functions specialized, and their reduced arguments are written to the previrt::proto::ComponentInterfaceTransform serialized file.
283 
284  Note: The original functions are not removed from the Module, unless \e 'Poccam' Pass is applied to perform dead code elimination.
285  */
286  virtual bool
287  runOnModule(Module& M)
288  {
289  if (transform.interface == NULL) return false;
290 
291  std::list<Function*> to_add;
292  CallGraph& CG = getAnalysis<CallGraph> ();
293  SpecializationPolicy* policy = SpecializationPolicy::recursiveGuard(
294  SpecializationPolicy::aggressivePolicy(), CG);
295 
296  bool modified = SpecializeComponent(M, this->transform, *policy, to_add);
297 
298  Module::FunctionListType &functionList = M.getFunctionList();
299  while (!to_add.empty())
300  {
301  Function *add = to_add.front();
302  to_add.pop_front();
303  if (add->getParent() == &M)
304  {
305  // Already in module
306  continue;
307  }
308  else
309  {
310  functionList.push_back(add); // "functionList" is the Module list of functions. push_back adds the function to the module
311  }
312  }
313 
314  if (SpecializeComponentOutput != "") // If the output filename is supplied as an argument, serialize the ComponentInterfaceTransform to the file
315  {
316  proto::ComponentInterfaceTransform buf; // Google protocol buffer
317  codeInto(this->transform, buf);
318  std::ofstream output(SpecializeComponentOutput.c_str(), std::ios::binary | std::ios::trunc); // Serializing buffer to file
319  assert(buf.SerializeToOstream(&output));
320  output.close();
321  }
322 
323  policy->release();
324 
325  return modified;
326  }
