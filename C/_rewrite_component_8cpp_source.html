104  /*! The function rewrites the call sites to functions present in the previrt::ComponentInterfaceTransform.
105  The function performs the following sequence of operations:
106 
107  1) Iterates over the previrt::CallRewrite objects in the previrt::ComponentInterfaceTransform object.
108 
109  2) If the previrt::CallRewrite object applies to a function not defined in the current Module simply skip.
110 
111  3) If the function is found within the module, iterate over all of the uses of the function and visit all the call-sites to the function.
112 
113  4) If a previrt::CallRewrite is available for the particular call-site, rewrite the call-site with fewer arguments.
114 
115  5) Replace the original call-site with the rewritten call-site
116 
117  6) If any single call-site was modified, the function returns true, otherwise returns false.
118  */
119  bool
120  TransformComponentWithUse(Module& M, ComponentInterfaceTransform& T)
121  {
122  bool modified = false;
123 
124  // Iterating over each tuple of rewrite that has been written to the ComponentInterfaceTransform buffer
125  for (ComponentInterfaceTransform::FMap::const_iterator i = T.rewrites.begin(), e = T.rewrites.end(); i != e; ++i)
126  {
127  Function* f = M.getFunction(i->first); // This is merely the prototype of the function (function declaration), that has to exist, if an external function
128  // is called
129  if (f == NULL) continue;
130 
131  // Iterate over the usages of the external function
132  for (Function::use_iterator ui = f->use_begin(), ue = f->use_end(); ui != ue; ++ui)
133  {
134  User* U = *ui;
135  if ((!isa<CallInst>(U) && !isa<InvokeInst>(U))
136  || !CallSite(cast<Instruction>(U)).isCallee(ui))
137  {
138  // Not a call, or being used as a parameter rather than as the callee.
139 
140  }
141  else
142  {
143  // The instruction is a call site
144  CallSite cs(cast<Instruction>(U));
145  const CallRewrite* const rw = T.lookupRewrite(i->first, cs.arg_begin(), cs.arg_end()); // If a rewrite does not exist
146  if (rw == NULL) continue; // skip this iteration
147 
148  // Debugging --Writing diagnostic messages to the stderr stream
149 #if DUMP
150  BasicBlock* owner = cs.getInstruction()->getParent();
151  errs() << "Specializing (inter-module) call to '" << cs.getCalledFunction()->getName()
152  << "' in function '" << (owner == NULL ? "??" : owner->getParent()->getName())
153  << "' on arguments [";
154  for (unsigned int i = 0, cnt = 0; i < cs.arg_size(); ++i)
155  {
156  if (!vector_in(rw->args, i))
157  {
158  if (cnt++ != 0)
159  errs() << ",";
160  errs() << i << "=(" << *cs.getArgument(i) << ")";
161  }
162  }
163  errs() << "]\n";
164 #endif
165 
166  Instruction* newInst = applyRewriteToCall(M, rw, cs); //Specialize the callsite
167  llvm::ReplaceInstWithInst(cs.getInstruction(), newInst); // Replace the old call with the new specialized call
168 
169  modified = true;
170  }
171  }
172  }
173 
174  return modified;
175  }

273  /*! RewriteComponentPass implements one phase of a three phase Inter-Module specialization process.
274  The steps of the process are:
275 
276  1) Gathering external references across all modules, so that the specific usages of functions outside the modules where they
277  are defined can be captured. This collection is performed by the 'Pinterface2' pass implemented, in the previrt::GatherInterface2Pass
278  class.
279 
280  2) Given the specific usages, serialized into a file as a previrt::proto::ComponentInterfaceTransform object, the previrt::SpecializeComponentPass
281  creates specialized versions of function in the modules where they have been defined.
282 
283  3) Once the specialized functions are created, the corresponding calls to the specialized versions, need to be rewritten,
284  taking a fewer number of arguments. The call rewriting process is accomplished by the \e 'Prewrite' Pass implemented by the
285  RewriteComponentPass class.
286 
287  RewriteComponentPass is registered as a Pass, with the PassManager internal database by the name 'Prewrite'.
288  The Pass takes an argument (required argument) 'Prewrite-input' flag followed by a file name that specifies
