63  /*! Returns true if \e 'r' is at least as strict a specialization policy as \e 'l'. The ConstSpecScheme
64  array, contains \e NULL values against arguments that are not be specialized and the values of the arguments to the specialized.
65  \e 'r' is stricter than \e 'l', if \e 'r' has at-least as many NULL values as \e 'l', and whenever and index in \e 'r' is NON-NULL (a value for an argument)
66  ,it matches the corresponding value in 'l' ConstSpecScheme.
67  */
68  bool Specialization::refines(int len, ConstSpecScheme l, ConstSpecScheme r)
69  {
70  for (int i = 0; i < len; i++) {
71  if (r[i] == NULL)
72  continue;
73  if (r[i] == l[i])
74  continue;
75  return false;
76  }
77  return true;
78  }

81  bool Specialization::refinesExt(const SmallBitVector &wrt, ConstSpecScheme l, ConstSpecScheme r)
82  {
83  /// wrt is the set of indices included in r, e.g.
84  /// wrt = [1,0,1,0,0] means:
85  /// args 0,1,3 are NOT included in r
86  /// args 2,3 are included in r
87  const int len = wrt.size();
88  for (int lp = 0, rp = 0; lp < len; lp++) {
89  if (wrt[lp]) {
90  if (r[rp] == NULL || l[lp] == r[rp])
91  rp++;
92  else
93  return false;
94  }
95  }
96  return true;
97  }

115  /*! The meta data can contain a list of specializations (names of original functions and specialized versions). If any specialization meta data
116  is available the initialize routine fills in the Specialization Table with the data.
117  */
118  void
119  SpecializationTable::initialize(Module* m)
120  {
121  assert(this->module == NULL);
122  this->module = m;
123 #ifdef RECORD_IN_METADATA
124  // Parse the module meta data to populate the table
125  NamedMDNode* specs = m->getNamedMetadata("previrt::specializations");
126  if (specs == NULL)
127  return;
128 
129  errs() << "Specialization Count: " << specs->getNumOperands() << "\n";
130 
131  for (unsigned int i = 0; i < specs -> getNumOperands(); ++i)
132  {
133  MDNode* funNode = specs->getOperand(i);
134  if (funNode == NULL)
135  {
136  continue;
137  }
138 
139  assert (funNode->getNumOperands() > 2);
140  MDString* prinName = dyn_cast_or_null<MDString>(funNode->getOperand(0));
141  MDString* specName = dyn_cast_or_null<MDString>(funNode->getOperand(1));
142 
143  if (prinName == NULL || specName == NULL)
144  {
145  errs() << "must skip " << (prinName == NULL ? "?" : prinName -> getString()) << "\n";
146  continue;
147  }
148 
149  Function* prin = m -> getFunction(prinName -> getString());
150  Function* spec = m -> getFunction(specName -> getString());
151 
152  if (prin == NULL || spec == NULL)
153  {
154  errs() << "must skip " << (prin == NULL ? "?" : prin -> getName()) << "\n";
155  continue;
156  }
157 
158  const unsigned int arg_count = prin -> getArgumentList().size();
159  if (funNode->getNumOperands() != 2 + arg_count)
160  {
161  continue;
162  }
163 
164  SpecScheme scheme = new Value*[arg_count];
165  for (unsigned int i = 0; i < arg_count; i++)
166  {
167  Value* opr = funNode->getOperand(2 + i);
168  if (opr == NULL)
169  {
170  scheme[i] = NULL;
171  }
172  else
173  {
174  assert (dyn_cast<Constant>(opr) != NULL);
175  scheme[i] = opr;
176  }
177  }
178 
179  this->addSpecialization(prin, scheme, spec, false);
180  errs() << "recording specialization of '" << prin->getName() << "' to '" << spec->getName() << "'\n";
181  }
182 #endif /* RECORD_IN_METADATA */
183  }

247  /*! Returns all the specializations against a function that are at least as strict as specified by the \e scheme (calling
248  SpecializationTable::getSpecializations).
249  A specialization is as strict as the other if it specializes no more arguments as compared to the other.
250  For example, consider a function with three arguments, and the \e scheme specifies only the first argument as specialize-able (NON-NULL value),
251  then all specializations that specialize the first argument, or the ones specializing none are returned.
252  Essentially we want only the matching specializations, therefore in the files StrategySpecializer.cpp line 167,
253  SpecializationTable::refines is called once again this time with inverting the specScheme (to specialize with respect to), and
254  the specialization scheme of the arguments, thereby only functions that take exactly the same arguments can be extracted from the
255  Specialization Table
256  */
257  void
258  SpecializationTable::getSpecializations(Function* f, ConstSpecScheme scheme,
259  std::vector<const Specialization*>& result) const
260  {
261  const Specialization* current = this->getSpecialization(f);
262  const int arg_count = f->getArgumentList().size();
263 
264  // Iterate over all the specializations(children) of the function if there exists any. There might be no specialization as of yet
265  for (std::vector<Specialization*>::const_iterator i =
266  current->children.begin(), e = current->children.end(); i != e; ++i)
267  {
268  if (Specialization::refines(arg_count, (*i)->args, scheme))
269  {
270  // (*i)-> args contains the the scheme(array of arguments) for the specialization
271  // Only add the specialization to the result if it is stricter or equal to the specScheme specified as parameter
272  result.push_back(*i);
273  }
274  }

278  /*! The addSpecialization function creates a Specialization node against a function pointer.
279  Adds an entry for a specialization of a function with respect to a certain usage, so that if the function
280  is called again with the same usage, the once created specialized copy is used rather creating a new one.
281 
282  @param parent a pointer to the original function.
283  @param scheme argument specifying the specialization scheme, values for arguments to be specialized and NULL otherwise.
284  @param specialization specialized version of the function
285  @param record a boolean parameter indicating if meta data is to be added to the Module.
286  */
287  bool
288  SpecializationTable::addSpecialization(Function* parent,
289  ConstSpecScheme scheme, Function* specialization, bool record)
290  {
291  // "parent" is the callee function and "specialization" is the specialized version of the callee function
292  assert(parent != NULL);
293  assert(specialization != NULL);
294 
295  // SpecTable is a map of <Function *, specialization*> . The iterator is an iterator of the map.
296  // The find function is called to check if the specialized version is already in the map
297  SpecTable::iterator cursor = this->specialized.find(specialization);
298  // SpecTable is the typedef data type and specialized is a variable of type SpecTable
299 
300  //Specialization* spec = this->specialized.lookup(specialization->getName());
301  if (cursor != this->specialized.end()) // If the function is in the map
302  {
303  return false; // Return false because no need to add to the table AGAIN
304  }
305 
306  Specialization* parentSpec = this->getSpecialization(parent); // Given the function * get the specialization *
307 
308 
309  Specialization* spec = new Specialization();
310  spec->handle = specialization; // Save the specialized function *
311  spec->parent = parentSpec; // Pointer to the parent entry
312  spec->args = scheme; // Save the specialization scheme
313  this->specialized[specialization] = spec; // Add an entry of the specialized function in the map
314  parentSpec->children.push_back(spec); // Add the new Specialization just created to the list of children of the parent Specialization
315 
316 
317 
318 #if RECORD_IN_METADATA
319  // Add the meta-data
320  if (record)
321  {
322  NamedMDNode* md = this-> module-> getOrInsertNamedMetadata("previrt::specializations");
323 
324  const unsigned int arg_count = parent->getArgumentList().size();
325  Value** vals = new Value*[2 + arg_count];
326  vals[0] = MDString::get(parent->getContext(), parent->getName());
327  vals[1] = MDString::get(parent->getContext(), specialization->getName());
328  for (unsigned int i = 0; i < parent->getArgumentList().size(); ++i) {
329  vals[i+2] = scheme[i];
330  }
331 
332  ArrayRef<Value*> ar_vals(vals, vals+2+arg_count);
333  MDNode* nde = MDNode::get(this->module->getContext(), ar_vals);
334 
335  md->addOperand(nde);
336  }
337 #endif /* RECORD_IN_METADATA */
338 
339  return true;
340  }

359  return this->specialized[f];
