86  /*! runOnModule - Compute the call graph for the specified module. Adding each function in the Module
87  to the call graph, by calling function addToCallGraph that adds a function to the call graph, and links the node to all
88  of the functions that it calls. It also Adds the non call uses to ExternalCallingNode
89  */
90  virtual bool runOnModule(Module &M)
91  {
92  CallGraph::initialize(M);
93 
94  ExternalCallingNode = getOrInsertFunction(0);
95  CallsExternalNode = new CallGraphNode(0);
96  Root = 0;
97 
98  // Add every function to the call graph.
99  for (Module::iterator I = M.begin(), E = M.end(); I != E; ++I)
100  addToCallGraph(I);

102  // If we didn't find a main function, use the external call graph node
103  if (Root == 0) Root = ExternalCallingNode;
104 
105  return false;

179  /*! The function returns a call graph Node for the Location */
180  CallGraphNode*
181  getNodeForLocation(AliasAnalysis& aa, AliasAnalysis::Location& loc)
182  {
183  std::map<AliasAnalysis::Location, CallGraphNode*>::iterator i = this->locationNodes.find(loc);
184  if (i != this->locationNodes.end()) {
185  return i->second;
186  }
187  else {
188  // TODO What is the structure that I want here?
189  // I think I want a single call graph node that has calls to several nodes..
190  CallGraphNode* cgn = new CallGraphNode(NULL);
191 
192  // Iterating over all of the functions in the Module
193  for (Module::iterator i = this->Mod->begin(), e = this->Mod->end(); i != e; ++i) {
194  if (!aa.isNoAlias(loc, locationForValue(aa, &*i))) // isNoAlias: A trivial helper function to check to see if the
195  //specified pointers are no-alias
196  {
197  // if the function did not have an existing alias, add it to the call graph
198  cgn->addCalledFunction(CallSite(), getOrInsertFunction(&*i));
199  }
200  }
201 
202  return cgn;
203  }

206  /*! addToCallGraph - Add a function to the call graph, and link the node to all
207  of the functions that it calls. It also adds the non call uses to ExternalCallingNode
208  */
209  void addToCallGraph(Function *F)
210  {
211  CallGraphNode *Node = getOrInsertFunction(F);
212 
213  // If this function has external linkage, anything could call it.
214  if (!F->hasLocalLinkage())
215  {
216  ExternalCallingNode->addCalledFunction(CallSite(), Node);
217 
218  // Found the entry point?
219  if (F->getName() == "main")
220  {
221  if (Root) // Found multiple external mains? Don't pick one.
222  Root = ExternalCallingNode;
223  else
224  Root = Node; // Found a main, keep track of it!
225  }
226  }
227 
228  //! Loop over all of the users of the function, looking for non-call uses.
229  for (Value::use_iterator I = F->use_begin(), E = F->use_end(); I != E; ++I)
230  {
231  User *U = *I;
232  if ((!isa<CallInst>(U) && !isa<InvokeInst>(U))
233  || !CallSite(cast<Instruction>(U)).isCallee(I))
234  {
235  // Not a call, or being used as a parameter rather than as the callee.
236  // The function being used as a parameter or function pointer may be used to make dynamic calls
237  ExternalCallingNode->addCalledFunction(CallSite(), Node);
238  break;
239  }
240  }
241 
242  // If this function is not defined in this translation unit, it could call
243  // anything.
244  if (F->isDeclaration() && !F->isIntrinsic())
245  Node->addCalledFunction(CallSite(), CallsExternalNode);
246  // All functions such as printf etc defined externally are added to CallsExternalNode
247 
248  AliasAnalysis* aa = this->getAnalysisIfAvailable<AliasAnalysis>();
249 
250  // Look for calls by this function.
251  for (Function::iterator BB = F->begin(), BBE = F->end(); BB != BBE; ++BB)
252  for (BasicBlock::iterator II = BB->begin(), IE = BB->end(); II != IE; ++II)
253  {
254  CallSite CS(cast<Value>(II));
255  if (CS && !isa<IntrinsicInst>(II))
256  {
257  const Function *Callee = CS.getCalledFunction();
258  if (Callee)
259  Node->addCalledFunction(CS, getOrInsertFunction(Callee)); // Add the function and the call-site to the Node
260  else
261  {
262  if (aa)
263  {
264  AliasAnalysis::Location loc = locationForValue(*aa, (Value*)Callee); // Creating a location for the callee function
265  Node->addCalledFunction(CS, getNodeForLocation(*aa, loc)); // Link the Node for the callee to the Node of the caller
266  }
267  else
268  Node->addCalledFunction(CS, CallsExternalNode);
269  }
270  }
271  }
272  }

275  //! destroy - Release memory allocated for the call graph
276  virtual void destroy()
277  {
278  /// CallsExternalNode is not in the function map, delete it explicitly.
279  if (CallsExternalNode)
280  {
281  CallsExternalNode->allReferencesDropped();

288  {
289  i->second->allReferencesDropped();
290  delete i->second;
291  }
292  locationNodes.clear();
293  }
294  CallGraph::destroy();
295  }
