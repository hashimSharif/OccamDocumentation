62  /*! Returns the number of matching arguments, inside the CallInfo structure and the arguments with their iterators sent as parameters.
63  If any single argument does not match or the number of arguments is not equal, the function returns NO_MATCH
64 
65  @param op_iterator begin iterator pointing to the start of the list of arguments.
66  @param op_iterator end iterator pointing to the end of the list of arguments.
67  */
68  int
69  CallInfo::refines(llvm::User::op_iterator begin, llvm::User::op_iterator end)
70  {
71  int matched = 0;
72 
73  std::vector<PrevirtType>::const_iterator from = this->args.begin(), to =
74  this->args.end();
75  for (; begin != end; ++begin, ++from)
76  {
77  if (from == to)
78  return NO_MATCH;
79 
80  int r = from->refines(begin->get());
81 
82  if (r == NO_MATCH)
83  return NO_MATCH;
84 
85  matched += r;
86  }
87  if (from != to)
88  return NO_MATCH;
89 
90  return matched;
91  }

183  /*! The function adds a call to the ComponentInterface, the call information includes the name of the call, and the set of arguments
184  The calls are logged in the \e 'Pinterface2' Pass implemented in the class previrt::GatherInterface2Pass
185  */
186  void
187  ComponentInterface::call(FunctionHandle f, User::op_iterator args_begin,
188  User::op_iterator args_end)
189  {
190  if (this->calls.find(f) == this->calls.end()) {
191  std::vector<CallInfo*> calls;
192  calls.push_back(CallInfo::Create(args_begin, args_end, 1));
193  this->calls[f] = calls;
194  } else {
195  std::vector<CallInfo*>& calls = this->calls[f];
196  for (std::vector<CallInfo*>::const_iterator begin = calls.begin(), end =
197  calls.end(); begin != end; ++begin) {
198  User::op_iterator cur = args_begin;
199  for (std::vector<PrevirtType>::const_iterator i =
200  (*begin)->args.begin(), e = (*begin)->args.end(); i != e; ++i) {
201  if (i->refines(cur->get()) == NO_MATCH)
202  goto no;
203  }
204  (*begin)->count++;
205  return;
206  no: continue;
207  }
208 
209  this->calls[f].push_back(CallInfo::Create(args_begin, args_end, 1));
210  }

254  CallInfo*
255  ComponentInterface::getOrCreateCall(FunctionHandle f, const std::vector<
256  PrevirtType>& args)
257  {
258  FunctionIterator itr = calls.find(f);
259  CallInfo* result;
260  if (itr == calls.end()) {
261  std::vector<CallInfo*> infos;
262  result = CallInfo::Create(args, 0);
263  infos.push_back(result);
264  calls[f] = infos;
265  return result;
266  }
267  else
268  {
269  for (CallIterator c = itr->second.begin(), e = itr->second.end(); c != e; ++c) {
270  if (args.size() != (*c)->args.size())
271  continue;
272  std::vector<PrevirtType>::const_iterator x = args.begin();
273  for (std::vector<PrevirtType>::const_iterator y = (*c)->args.begin(),
274  z = args.end(); x != z; ++x, ++y) {
275  if (*x != *y)
276  break;
277  }
278  if (x == args.end()) {
279  return *c;
280  }
281  }
282  result = CallInfo::Create(args, 0);
283  calls[f].push_back(result);
284  return result;
285  }

633  /*! Given the functionHandle to the original function and a set of specific arguments, a corresponding CallRewrite is returned if one
634  exists in the previrt::ComponentInterfaceTransform object. If a CallRewrite does not exist for the function with the specific
635  set of arguments, the function returns \e NULL

638  @param op_begin start of the argument iterator
639  @param ob_end end of the argument iterator

641  */

643  ComponentInterfaceTransform::lookupRewrite(FunctionHandle name,
644  llvm::User::op_iterator op_begin, llvm::User::op_iterator op_end) const
