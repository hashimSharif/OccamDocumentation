214  /// argRangeAdd adds a subset of \e args from the valueEnv vector to the result vector specified by the indices \e from and \e to
215  bool
216  argRangeAdd(int from, int to, std::vector<Value*>& result) const
217  {
218  // Adjusting negative indexes
219  if (from < 0)
220  from += argCount;
221  if (to < 0)
222  to += argCount;
223 
224  if (from < 0 || from >= argCount || to < 0 || from >= argCount) { // Invalid indeces from and to
225  return false;
226  }
227  if (from < to) {
228  // Prototype of vector::insert :void insert (iterator position, InputIterator first, InputIterator last);
229  result.insert(result.end(), valueEnv.begin() + from, valueEnv.begin() // Adds the arguments to the vector 'result'
230  + to);
231  } else {
232  result.insert(result.end(), valueEnv.rbegin() + (argCount - from),
233  valueEnv.rbegin() + (argCount - to));

259  /*! The previrt::getCallFunction returns the pointer to a Function, that was pushed into the WatchEnv::matchEnv array with the call
260  to previrt::patternMatches. The function pushed was the one that matched the expression in the watch::proto::patternExpr object.
261  */
262  Value*
263  WatchContext::getCallFunction(WatchEnv& env, const google::protobuf::RepeatedPtrField<watch::proto::PrimExpr>& exprs,
264  const std::vector<Type*>& types)
265  {
266 
267  if (exprs.size() == 0)
268  {
269  errs() << "No target given!\n";
270  return NULL;
271  }
272 
273  // Special case for calling variable
274  if (exprs.size() == 1)
275  {
276  const watch::proto::PrimExpr& expr = exprs.Get(0); // Get the expression from the list of expressions
277  FunctionType* type = NULL;
278  // FunctionType object holds the parameters to the function, the type of each parameter, and the return type of the function
279 
280  if (expr.has_var()) // Is var member set (Assigned any value)
281  {
282  Value* v = env.lookupValue(expr.var()); // Searching the list of arguments in the WatchEnv variable given the index
283  if (v == NULL)
284  {
285  return NULL;
286  }
287  type = dyn_cast<FunctionType> (v->getType()); // type will be assigned non null only if v is a function
288  if (type == NULL)
289  {
290  errs() << "Trying to call non-function!\n";
291  return NULL;
292  }
293 
294  return v;
295  }
296  }
297 
298  // The following block of code executes when the size of expressions is greater than 1
299  // The normal case is a string that is constructed from components
300  Twine name;
301  for (google::protobuf::RepeatedPtrField<watch::proto::PrimExpr>::const_iterator i = exprs.begin(), e = exprs.end(); i != e; ++i)
302  {
303  if (i -> has_var() || i -> has_lit())
304  {
305  return NULL;
306  }
307  if (i -> has_match())
308  {
309  std::string res;
310  // In the call to lookupMatch , i->match is an integer index and res is the string to retrieve the result string
311  if (!env.lookupMatch(i->match(), res)) { // search in the list of global variables. res contains the name of the global
312  errs() << "didn't find! " << i->match() << "\n";
313  return NULL;
314  }
315  name = name.concat(res);
316  }
317  else if (i -> has_str())
318  {
319  // Get the string...
320  name = name.concat(i->str());
321  }
322 
323  }
324 
325  std::string fullName = name.str(); // Get the string
326  Constant* f = this -> module -> getFunction(fullName);
327 
328  if (f != NULL) // If the function was found in the module
329  return f;
330 
331  // If the function was not found in the module, find it in the list of added_functions
332  for (std::list<Function*>::iterator i = added_functions.begin(), e = added_functions.end(); i != e; ++i)
333  {
334  if (fullName == (*i)->getName().str())
335  {
336  errs() << "early match '" << fullName << "' '" << (*i)->getName().str() << "'\n";
337  return *i;
338  }
339  }
340 
341  // Create a new function if one does not exist and add it to the list 'added_functions'
342  ArrayRef<Type*> args(types);
343  FunctionType* type = FunctionType::get(Type::getVoidTy(getContext()), args, false);
344 
345  Function* function = Function::Create(type, GlobalVariable::ExternalWeakLinkage, fullName.c_str()); // Create the function with weak linkage so that a more definitive
346  added_functions.push_back(function); // definition is allowed to take over
347 
348  return function; // return the function
349  }

366  exists do not create a new one. */
367  static Constant*
368  getExit(WatchContext* ctx, const char* failureName, bool fancy)
369  {
370  Function* failureFunction = ctx->module->getFunction(failureName);
371  if (failureFunction != NULL) // If the failure function exists in the module
372  {
373  FunctionType* typ = failureFunction->getFunctionType();
374  bool badType = false;
375  if (fancy)
376  {
377  if (!typ->isVarArg() || !typ->getReturnType()->isVoidTy()
378  || typ->getNumParams() != 1 || !typ->getParamType(0)->isPointerTy())
379  {
380  PointerType* pty = dyn_cast<PointerType> (typ->getParamType(0));
381  if (!pty->getElementType()->isIntegerTy(8))
382  {
383  badType = true;
384  }
385  }
386  else
387  {
388  badType = true;
389  }
390  }
391  else // if fancy is false
392  {
393  badType = typ->isVarArg() || !typ->getReturnType()->isVoidTy()
394  || typ->getNumParams() != 1 || !typ->getParamType(0)->isIntegerTy(
395  32);
396  }
397 
398  if (badType) {
399  errs() << "Failure function already exists with bad function type.\n";
400  typ->dump();
401  return NULL;
402  } else
403  return failureFunction;
404  }
405  else // If the failure function doesn't exist, create a new and add it to the list "added_functions"

410  Type*> (Type::getInt8PtrTy(ctx->getContext(), 0)), true);
411  } else {
412  typ = FunctionType::get(Type::getVoidTy(ctx->getContext()), ArrayRef<
413  Type*> (Type::getInt32Ty(ctx->getContext())), false);
414  }
415  failureFunction = Function::Create(typ, GlobalVariable::ExternalLinkage,
416  failureName);
417  ctx->added_functions.push_back(failureFunction);
418  failureFunction->setDoesNotReturn(true);

423  static Value*
424  getValue(WatchEnv& env, IRBuilder<>& irb, const watch::proto::PrimExpr& expr)
425  {
426  if (expr.has_var()) {
427  return env.lookupValue(expr.var());
428  }
429  else if (expr.has_lit()) {
430  if (expr.lit().type() == proto::I)
431  {
432  Type* type = IntegerType::get(env.getContext(),
433  expr.lit().int_().bits());
434  Module& m = *irb.GetInsertBlock()->getParent()->getParent();
435  return PrevirtType(expr.lit()).concretize(m, type);
436  }
437  else {
438  // TODO: literals not implemented!
439  errs() << "Only integer literals are implemented!\n";
440  return NULL;
441  }
442  }
443  else if (expr.has_match()) {
444  return env.materializeMatch(irb, expr.match());
445  }
446  else if (expr.has_str()) {
447  std::string contents = expr.str();
448  Constant* cnst = llvm::ConstantDataArray::getString(env.getContext(), expr.str(), true);
449 
450  ConstantInt* zero32 = ConstantInt::get(IntegerType::getInt32Ty(
451  env.getContext()), 0, false);
452  return ConstantExpr::getGetElementPtr(cnst, zero32, true);
453  }
454  else {
455  errs() << "Empty PrimExpr!\n";
456  return NULL;
457  }
458  }

461  /*! The getArguments function populates the result vector reading the indices array. */
462  static bool
463  getArguments(
464  WatchEnv& env,
465  IRBuilder<>& irb,
466  const google::protobuf::RepeatedPtrField<watch::proto::PrimExprs>& indicies,
467  std::vector<Value*>& result)
468  {
469  int count = 0;
470  for (google::protobuf::RepeatedPtrField<watch::proto::PrimExprs>::const_iterator
471  i = indicies.begin(), e = indicies.end(); i != e; ++i) {
472  if (i->has_one() && i->has_args()) {
473  errs() << "Too many PrimExprs!\n";
474  return false;
475  }
476  if (i->has_one()) { // If only one argument
477  count++;
478  continue;
479  }
480  if (i->has_args()) { //multiple arguments
481  count += i->args().end() - i->args().start();
482  continue;
483  }
484 
485  errs() << "Empty PrimExprs!\n";
486  return false;
487  }
488  result.reserve(count);
489 
490  for (google::protobuf::RepeatedPtrField<watch::proto::PrimExprs>::const_iterator
491  i = indicies.begin(), e = indicies.end(); i != e; ++i)
492  {
493  if (i->has_one()) {
494  Value* val = getValue(env, irb, i->one());
495  result.push_back(val);
496  }
497  else
498  {
499  assert(i->has_args());
500 
501  if (!env.argRangeAdd(i->args().start(), i->args().end(), result)) { // Add the arguments in the PrimExpr to the result vector
502  return false;
503  }
504  }
505 
506  }
507 
508  return true;
509  }

541  /*!
542  The compileAction function uses the IRBuilder to construct, a wrapper function over the original function to watch the function.
543  (Enforce the interface specified by the Google protocol buffer serialized file). It compares the values of the arguments to be
544  equal to the previrt::PrevirtType values. If found unequal a failure function is called. If the interface is not violated by a call,
545  the original function (delegate) is invoked. The compileAction Function is called recursively, as the action policy can be nested
546  The 'if' actions are the only actions can be nested.
547  */
548  static bool
549  compileAction(Function* inside, const Function* const delegate,
550  const watch::proto::ActionTree& policy, WatchContext& ctx, WatchEnv& env,
551  BasicBlock* result, BasicBlock* failure, BasicBlock* success)
552  {
553  IRBuilder<> builder(result); // Adding IR to the basic block sent as parameter
554 
555  if (policy.has_if_())
556  {
557  PrevirtType pt(policy.if_().test()); // policy.if_().test() is a PrevirtType object
558  Function* eq = pt.getEqualityFunction(inside->getParent());
559 
560  if (eq == NULL)
561  {
562  errs() << "No equality function, defaulting to false\n";
563  if (policy.if_().has__else())
564  {
565  return compileAction(inside, delegate, policy.if_()._else(), ctx,
566  env, result, failure, success);
567  }
568  else
569  {
570  errs()
571  << "No equality function, and no else. defaulting to failure\n";
572  builder.CreateBr(failure); // Add an unconditional branch to the failure basic block
573  return true;
574  }
575 
576  }
577  else // Equality function does exist
578  {
579  Value* compLeft = env.lookupValue(policy.if_().var()); // An argument specified by policy.if()_.var()
580  Value* compRight = pt.concretize(*inside->getParent(), compLeft->getType()); // The previrtType to compare the actual argument against
581  env.valueEnv[policy.if_().var()] = compRight;
582  BasicBlock* _then = BasicBlock::Create(result->getContext());
583  if (!compileAction(inside, delegate, policy.if_()._then(), ctx, env,
584  _then, failure, success))
585  {
586  return false;
587  }
588 
589  env.valueEnv[policy.if_().var()] = compLeft;
590  BasicBlock* _else;
591 
592  if (policy.if_().has__else())
593  {
594  _else = BasicBlock::Create(result->getContext());
595 
596  if (!compileAction(inside, delegate, policy.if_()._else(), ctx, env,
597  _else, failure, success))
598  {
599  return false;
600  }
601  }
602  else
603  {
604  _else = success; // If no else block exists point to the success block given as parameter to the function
605  }
606 
607  // Adding call to the equality function
608  std::vector<Value*> argVals;
609  argVals.reserve(2);
610  argVals.push_back(compLeft);
611  argVals.push_back(compRight);
612  Value* test = builder.CreateCall(eq, ArrayRef<Value*> (argVals)); // Create a call to the equality function
613  SwitchInst* sw = builder.CreateSwitch(test, _else, 1); // The return value of the call to the function "eq" is stored in test on which switch is applied
614  sw->addCase(ConstantInt::getTrue(ctx.getContext()), _then);
615  return true;
616  }
617 
618  }
619  else if (policy.has_forward()) {
620 
621  CallInst* call = builder.CreateCall(const_cast<Function*> (delegate), // Create a call to the function "delegate"
622  ArrayRef<Value*> (env.valueEnv.data(), env.valueEnv.data() //valueEnv.data() return the pointer to the first element of the vector
623  + delegate->getArgumentList().size()));
624  env.valueEnv.push_back(call);
625  builder.CreateBr(success);
626 
627  return true;
628  }
629  else if (policy.has_call())
630  {
631 
632  std::vector<Value*> args;
633  BasicBlock* def = BasicBlock::Create(ctx.getContext(), "defined", inside);
634  IRBuilder<> dbuilder(def);
635 
636  if (!getArguments(env, dbuilder, policy.call().args(), args))
637  {
638  errs() << "index out of bounds\n";
639  return false;
640  }
641 
642  Value* evt = ctx.getCallFunction(env, policy.call().target(), args); // The call is not yet added to any basic block
643  if (evt == NULL) {
644  return false;
645  }
646 
647  BasicBlock* join = BasicBlock::Create(ctx.getContext(), "join", inside);
648  Value* evt_int = builder.CreatePtrToInt(evt, Type::getInt32Ty(
649  ctx.getContext()));
650  SwitchInst* sw = builder.CreateSwitch(evt_int, def, 1); // The switch is created in the result block, and as a case jumps to "def"
651  sw->addCase( ConstantInt::get(Type::getInt32Ty(ctx.getContext()), 0, true), join);
652 
653  CallInst* call = dbuilder.CreateCall(evt, ArrayRef<Value*> (args)); // The call is added to the "def" basic block
654  if (policy.call().tail())
655  {
656  tailCall(call, dbuilder);
657  return true;
658  }
659  else
660  {
661  dbuilder.CreateBr(join); // Create an unconditional jump to the "join" basic block
662  }
663 
664  IRBuilder<> jbuilder(join);
665  jbuilder.CreateBr(success); // Create an unconditional branch to "success" basic block
666  return true;
667 
668  }
669  else if (policy.has_return_())
670  {
671  IRBuilder<> builder(result);
672  if (policy.return_().undef())
673  {
674  returnUndef(inside, builder);
675  return result;
676  }
677  else if (policy.return_().has_value())

859  matches[i].rm_eo - matches[i].rm_so));
