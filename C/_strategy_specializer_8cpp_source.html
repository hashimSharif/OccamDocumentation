64 /*! The trySpecializeFunction iterates over all of the instructions in the function, looking for call instructions.
65  If the call contains constant arguments, a specialized copy of the function corresponding to the arguments is
66  looked up in the previrt::SpecializationTable (storing a list of applied specializations).
67  If the table does not contain a version corresponding to the particular usage,
68  a new cloned copy of the function is created, with arguments directly evaluated in the body of the function, hence
69  fewer parameters to the function. The specialized version is added to the previrt::SpecializationTable for future reference.
70 */
71 static bool
72 trySpecializeFunction(Function* f, SpecializationTable& table, SpecializationPolicy* policy, std::list<Function*>& to_add)
73 {
74 
75  bool modified = false;
76 
77  // Iterating over all of the instructions in the function
78  for (Function::iterator bb = f->begin(), bbe = f->end(); bb != bbe; ++bb)
79  {
80  for (BasicBlock::iterator I = bb->begin(), E = bb->end(); I != E; ++I)
81  {
82  CallSite call;
83 
84  if (CallInst* ci = dyn_cast<CallInst>(&(*I)))
85  {
86  call = CallSite(ci); // cast to call site
87  }
88  else if (InvokeInst* ii = dyn_cast<InvokeInst>(&(*I)))
89  {
90  call = CallSite(ii); // cast to call site
91  }
92  else
93  {
94  continue; // We are only interested in calls and invokes.
95  }
96 
97  // The code below is executed only if the instruction is a call or an invoke
98 
99  Function* callee = call.getCalledFunction(); // Get the pointer to the function being called
100  if (callee == NULL)
101  {
102  // dynamic call, can't resolve, calls using function pointers etc
103  continue;
104  }
105  const unsigned int callee_arg_count = callee->getArgumentList().size(); // get the number of arguments in the callee
106  if (callee == NULL || !canSpecialize(callee) || callee->isVarArg())
107  continue; // If the callee cannot be specialized don't care
108 
109 
110  if (callee->hasFnAttr(Attribute::NoInline)) //Q! Why are we skipping functions that cannot be in-lined?
111  {
112  errs() << "Function '" << callee->getName() << "' has no-inline, skipping.\n";
113  continue;
114  }
115 
116  if (callee->getName().equals(""))
117  {
118  errs() << "Skipping function with no name.\n";
119  continue;
120  }
121 
122  Value* const * specScheme = policy->specializeOn(callee, call.arg_begin(), call.arg_end());
123  // A specialization policy is being applied given the called function, and the arguments specified in the call
124  // The specScheme array has a specialization policy for each and every argument in the call
125  // This specialization scheme is just relevant to the call site because the arguments are taken from the call, not function
126 
127  if (specScheme == NULL) // In case there is nothing to specialize skip one iteration
128  continue; // The callee might have been a recursive function, or the arguments were not constants
129 
130  // == BEGIN DEBUGGING ============================================= NOT Relevant to program LOGIC !
131 #if DUMP
132  errs() << "Specializing call to '" << callee->getName() << "' in function '" << f->getName() << "' on arguments [";
133 
134  //-- Looping over the arguments in the callee function
135  for (unsigned int i = 0, cnt = 0; i < callee->getArgumentList().size(); ++i)
136  {
137  if (specScheme[i] != NULL)
138  {
139  if (cnt++ != 0) // If it's not the very first argument
140  errs() << ",";
141 
142  Value* v = call.getInstruction()->getOperand(i);
143  if (GlobalValue* gv = dyn_cast<GlobalValue>(v))
144  {
145  errs() << i << "=(@" << gv->getName() << ")";
146  }
147  else
148  {
149  errs() << i << "=(" << *call.getInstruction()->getOperand(i) << ")";
150  }
151  }
152  }
153  errs() << "]\n";
154 #endif
155  // == END DEBUGGING ===============================================
156 
157 
158  // Specialization * are a wrapper structure over function* , It is defined in SpecializationTable.cpp
159  std::vector<const SpecializationTable::Specialization*> versions;
160  table.getSpecializations(callee, specScheme, versions); // Get all versions of specialization for this callee and specScheme
161 
162  /* ----- Trying to find the specialized version in the SpecializationTable, if one exists ------ */
163 
164  Function* specializedVersion = NULL;
165  // Traversing each and every version of specialization for the pair
166  for (std::vector<const SpecializationTable::Specialization*>::iterator i = versions.begin(), e = versions.end(); i != e; ++i)
167  {
168  if (SpecializationTable::Specialization::refines(callee_arg_count, specScheme, (*i)->args))
169  {
170  // The above statement is only true if specScheme is exactly the same as (*i) -> args Because the getSpecializations function only returned those specializations which were stricter or equal to the specScheme
171  specializedVersion = (*i)->handle; // This returns a pointer to the specialized function if one exists
172  break;
173  }
174  }
175 
176  /* The specialized version wasn't found in the SpecializationTable, therefore we create a new version and add it to the Table */
177  if (NULL == specializedVersion)
178  {
179 
180  specializedVersion = specializeFunction(callee, specScheme);
181  table.addSpecialization(callee, specScheme, specializedVersion); // Added to table for further reference
182  // callee is the original function (still not removed by dead code elimination), specialized version is the version specialized
183  // according to specScheme
184 
185  to_add.push_back(specializedVersion); // Add the new function version to the list of functions "to_add"
186  }
187 
188 
189  assert(specializedVersion != NULL); // The specialized version cannot possibly be null by now
190  const unsigned int specialized_arg_count =
191  specializedVersion->getArgumentList().size();
192 
193 
194  /*---- Process of specializing all call-sites that correspond to the newly created specialized version of the function -----*/
195 
196  std::vector<unsigned> argPerm; // A vector containing integer values
197  argPerm.reserve(specialized_arg_count);
198  for (unsigned from = 0; from < callee_arg_count; from++)
199  {
200  if (specScheme[from] == NULL) // This is the case when the argument is to be removed, As it's value is directly added to function
201  argPerm.push_back(from); // Adding indexes to arguments that have been specialized
202  }
203  assert(specialized_arg_count == argPerm.size());
204 
205  // As the function has been specialized , the call site also needs to be specialized
206  Instruction* newInst = specializeCallSite(I, specializedVersion, argPerm);
207  llvm::ReplaceInstWithInst(bb->getInstList(), I, newInst); // replacing the original call site with the new call site in the program
208 
209  modified = true;
210  }
211  }
212  return modified;
213 }

216 /*! The runOnModule function performs the following set of operations to implement Partial evaluation:
217 
218  1) Iterates over all of the functions in the module calling trySpecializeFunction(defined in StrategySpecializer.cpp line 72)
219  on each function.
220 
221  2) The trySpecializeFunction routine analyses all the call-sites in the function. If any call-sites involve constant arguments, a new specialized
222  version of the function is created.
223 
224  3) Using the cloneFunction routine (llvm), the constants are embedded directly into the body of the specialized version, thereby
225  reducing the arguments to the function.
226 
227  4) The newly created versions are optimized using FunctionPassManager (llvm), to perform optimizations such as constant folding,
228  hence reducing possible number of control paths.
229 
230  5) The trySpecializeFunction is called again on the specialized version of the function, as it is likely that the optimized
231  function with constant replacing variables, could lead to further functions that could be specialized.
232 
233  6) The specialized functions each with a unique name assigned by previrt::specializeName are added to the Module.
234 
235  7) The previrt::specializeCallSite function is called to replace respective call instructions, with the new call instructions to specialized versions taking fewer arguments.
236 
237 */

239 bool SpecializerPass::runOnModule(Module &M)
240 {
241  // getAnalysis function is used by subclasses to get to the analysis information that they claim to use
242  // by overriding the getAnalysisUsage function
243  CallGraph& CG = this->getAnalysis<CallGraph> (); // This function is implemented inside ModulePass (llvm)
244  // Note CG is an alias /Reference to the original call graph
245 
246  // Perform SCC analysis
247  // (I can't seem to do this with the SCC pass because I'm not going to preserve it)
248  // TODO: I don't like hard-coding this
249  SpecializationPolicy* policy = SpecializationPolicy::recursiveGuard(
250  SpecializationPolicy::aggressivePolicy(), CG);
251 
252  std::set<Function*> empty_set; //Q! This empty set is not used At ALL ? Why ?
253 
254  std::list<Function*> to_add; // The new list of specialized functions
255  SpecializationTable table(&M); // A specialization table to hold all specializations
256 
257  FunctionPassManager* optimizer = NULL;
258 
259  if (this->optimize)
260  {
261  optimizer = new FunctionPassManager(&M);
262  PassManagerBuilder builder;
263  builder.OptLevel = 3;
264  builder.populateFunctionPassManager(*optimizer); // Sets level of optimization for the optimizer. The optimizer is used ahead
265  }
266 
267  bool modified = false;
268  for (Module::iterator f = M.begin(), fe = M.end(); f != fe; ++f)
269  {
270  modified = trySpecializeFunction(&*f, table, policy, to_add) || modified; // Once modified is set to true it remains true
271  }
272 
273  // A second pass of specialization is extremely important, so that the calls made within the specialized versions can also be
274  // analysed in order to create more specialized versions.
275 
276  while (!to_add.empty()) // Until all newly created functions have been added to the module
277  {
278  Function* f = to_add.front();
279  to_add.pop_front();
280 
281  if (f->getParent() == &M)
282  {
283  // The function was already in the module or
284  // has already been added in this round of
285  // specialization, no need to add it twice
286  continue; // skipping iteration
287  }
288 
289  if (optimizer)
290  {
291  optimizer->run(*f); // Optimize the cloned function. Possible optimizations can be made with the newly added constants
292  }
293 
294  // The trySpecializeFunction is called once again, as the newly cloned function is given constants that can be further specialize
295  // further call sites
296  trySpecializeFunction(f, table, policy, to_add);
297 
298  M.getFunctionList().push_back(f); // Add the specialized function to the module
299  }
300 
301  if (modified)
302  errs() << "...progress...\n";
303 
304  policy->release();
305  if (optimizer)
306  delete optimizer;
307 
308  return modified;
309 }
