82  /*!
83  * Remove all code from the given module that is not necessary to implement the given interface.
84  * In llvm each function is by default external-linkage(assumed to be called externally). To remove dead code it is
85  necessary to modify the linkage of the functions, not called externally to internal-linkage. previrt::localizeLinkage
86  is called to return an internal-linkage counter-part for each external definition. Once internalized the llvm PassManager
87  performs dead code elimination in the module. The dead code elimination is performed in a loop, until there is more code
88  still to be removed.
89 
90  The given Interface \e I of type previrt::proto::ComponentInterface is a list of functions and Globals that are accessed
91  from other modules. These functions and globals should not be internalized as they are being accessed from other modules.
92  */
93  bool
94  MinimizeComponent(Module& M, ComponentInterface& I)
95  {
96  bool modified = false;
97 
98  errs() << "interface!\n";
99  I.dump();
100 
101  // Set all functions that are not in the interface to internal linkage only
102  // Only by specifying internal linkage we specify that the functions and variables are to be
103  // used internally and therefore no assumption needs to be made of it being used from another compilation
104  // unit.
105  const StringMap<std::vector<CallInfo*> >::const_iterator end =
106  I.calls.end(); // Get the end of the Map. (used in further comparison)
107  for (Module::iterator f = M.begin(), e = M.end(); f != e; ++f)
108  {
109  // Specifying all functions in module to be internal linkage
110  if (!f->isDeclaration() && f->hasExternalLinkage() &&
111  I.calls.find(f->getName()) == end && // If the function Name was not found in the Map, and
112  I.references.find(f->getName()) == I.references.end()) // If the reference name was not found in the map, set to internal linkage
113  {
114  errs() << "Hiding '" << f->getName() << "'\n";
115  f->setLinkage(GlobalValue::InternalLinkage);
116  modified = true;
117  }
118  }
119 
120  for (Module::global_iterator i = M.global_begin(), e = M.global_end(); i != e; ++i)
121  {
122  // Specifying all globals to be internal linkage
123  if (i->hasExternalLinkage() && i->hasInitializer() &&
124  I.references.find(i->getName()) == I.references.end()) // If the reference to the global was not found in the map only then set
125  // to internal linkage
126  {
127  errs() << "internalizing '" << i->getName() << "'\n";
128  i->setLinkage(localizeLinkage(i->getLinkage()));
129  modified = true;
130  }
131  }
132  /* TODO: We want to do this, but libc has some problems...
133  for (Module::alias_iterator i = M.alias_begin(), e = M.alias_end(); i != e; ++i) {
134  if (i->hasExternalLinkage() &&
135  I.references.find(i->getName()) == I.references.end() &&
136  I.calls.find(i->getName()) == end) {
137  errs() << "internalizing '" << i->getName() << "'\n";
138  i->setLinkage(localizeLinkage(i->getLinkage()));
139  modified = true;
140  }
141  }
142  */
143 
144  //-- Perform global dead code elimination
145  // TODO: (SRI) To what extent should we do this here, versus
146  // doing it elsewhere?
147  PassManager cdeMgr, mfMgr, mcMgr;
148  cdeMgr.add(createGlobalDCEPass());
149  //mfMgr.add(createMergeFunctionsPass());
150  mcMgr.add(createConstantMergePass());
151  bool moreToDo = true;
152  unsigned int iters = 0;
153  while (moreToDo && iters < 10000)
154  {
155  moreToDo = false;
156  if (cdeMgr.run(M))
157  moreToDo = true;
158  //if (mfMgr.run(M)) moreToDo = true;
159  if (mcMgr.run(M))
160  moreToDo = true;
161  modified = modified || moreToDo;
162  ++iters;

168  //if (mfMgr.run(M)) errs() << "MergeFunctions still had more to do\n";
169  if (mcMgr.run(M)) errs() << "MergeConstants still had more to do\n";
170  }
171 
172  if (modified)
173  {
174  errs() << "...progress...\n";
175  }
176 
177  return modified;
178  }

187  /*! The OccamPass Class performs dead code elimination in the module. By default all functions in llvm are
188  external linkage(they can be called from an external module), hence it is necessary to be certain that a function is not called
189  from an external module before it is pruned. All functions not called externally, are modified to internal
190  linkage, and an llvm dead code elimination pass is applied.
191 
192  The OccamPass class is registered as a pass with the PassManager internal database with the name \e 'Poccam'. File names (required argument)
193  followed by the flag \e 'Poccam-input' are previrt::ComponentInterface objects that are a wrapper above the Google protocol buffer
194  previrt::proto:ComponentInterface that contain a list of functions and references that are being used externally.
195  The list of externally called functions and references is required so that they are not incorrectly eliminated from the module. */
196  class OccamPass : public ModulePass
197  {
198  public:

200  /// and a set of references (defining function names). Function names are required by the OccamPass class to correctly eliminate
201  /// all unused code across the modules.
202  ComponentInterface interface;

207  specifies the interface to prune with respect to.
