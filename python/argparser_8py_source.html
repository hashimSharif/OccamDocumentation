67  def parse_args(self,args):
68  '''Returns a triple (input_files, output_file, flags)'''
69  input_files = []
70  output_file = None
71  flags = []
72  i = 0
73  while i < len(args):
74  curFlag = self.extractFlag(args[i])
75  if curFlag is not None:
76  matchingShortOpts = filter(curFlag.startswith,self.shortWithOpt())
77  matchingShortOpts.sort(key=(lambda s: (-len(s), s)))
78  if curFlag in self.flags():
79  flags.append(args[i])
80  i += 1
81  elif len(matchingShortOpts) > 0:
82  shortOpt = matchingShortOpts[0]
83  if shortOpt == curFlag:
84  shortOptArg = args[i+1]
85  shortCount = 2
86  else:
87  shortOptArg = curFlag[len(shortOpt):]
88  shortCount = 1
89  if shortOpt == 'o':
90  output_file = shortOptArg
91  else:
92  flags.extend(args[i:i+shortCount])
93  i += shortCount
94  elif curFlag in self.longWithOpt():
95  if not '=' in args[i]:
96  logging.getLogger().error('ERROR: longWithOpt argument %s without \'=\'\n\twhile parsing %s',
97  args[i], args)
98  longOptArg = args[i][args[i].index('=')+1:]
99  if curFlag == 'output':
100  output_file = longOptArg
101  else:
102  flags.append(args[i])
103  i += 1
104  elif args[i].startswith('-') and args[i] != '-':
105  logging.getLogger().error('ERROR: unrecognized argument %s\n\twhile parsing %s',
106  args[i], args)
107  raise UnknownArgument(args[i], args)
108  else:
109  input_files.append(args[i])
110  i += 1
111  else:
112  if args[i].startswith('@'):

114  args[i], args)

116  else:

118  i += 1
