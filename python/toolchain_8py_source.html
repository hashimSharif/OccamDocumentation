87  def syms(s):
88  if not SCACHE.has_key(s):
89  SCACHE[s] = symbols(s)
90  return SCACHE[s]
91  DCACHE = {}
92  def defined(s):
93  if not DCACHE.has_key(s):

95  DCACHE[s] = ss['T'].union(ss['t']).union(ss['d']).union(ss['D']).union(ss['W']).union(ss['C'])
96  return DCACHE[s]
97  UCACHE = {}
98  def undefined(s):
99  if not UCACHE.has_key(s):
100  ss = syms(s)
101  UCACHE[s] = ss['U'].union(ss['u']).difference(defined(s))
102  return UCACHE[s]

116  undef_symbols = undef_symbols.union(undefined(x)).difference(defined(x))

120  progress = False
